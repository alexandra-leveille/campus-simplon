#installer vue-cli (dépend de nodejs + npm)

npm install -g vue-cli

# doc => https://github.com/vuejs/vue-cli

#démarrer un projet VueJS (avec un template)

vue init webpack-simple

# OU

vue init webpack my-project


# depuis le dossier créé pour le projet, installer les dépendances
npm install

# lancer le serveur de dev
npm run dev
